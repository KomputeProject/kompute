132 OpMult<tX, tY, tZ>::init()
133 {
134  SPDLOG_DEBUG("Kompute OpMult init called");
135 
136  if (this->mTensors.size() < 3) {
137  throw std::runtime_error(
138  "Kompute OpMult called with less than 1 tensor");
139  } else if (this->mTensors.size() > 3) {
140  spdlog::warn("Kompute OpMult called with more than 3 this->mTensors");
141  }
142 
143  this->mTensorLHS = this->mTensors[0];
144  this->mTensorRHS = this->mTensors[1];
145  this->mTensorOutput = this->mTensors[2];
146 
147  // The dispatch size is set up based on either explicitly provided template
148  // parameters or by default it would take the shape and size of the tensors
149  if (tX > 0) {
150  // If at least the x value is provided we use mainly the parameters
151  // provided
152  this->mX = tX;
153  this->mY = tY > 0 ? tY : 1;
154  this->mZ = tZ > 0 ? tZ : 1;
155  } else {
156  // TODO: Fully support the full size dispatch using size for the shape
157  this->mX = this->mTensorLHS->size();
158  this->mY = 1;
159  this->mZ = 1;
160  }
161  spdlog::info("Kompute OpMult dispatch size X: {}, Y: {}, Z: {}",
162  this->mX,
163  this->mY,
164  this->mZ);
165 
166  // TODO: Explore adding a validate function
167  if (!(this->mTensorLHS->isInit() && this->mTensorRHS->isInit() &&
168  this->mTensorOutput->isInit())) {
169  throw std::runtime_error(
170  "Kompute OpMult all tensor parameters must be initialised. LHS: " +
171  std::to_string(this->mTensorLHS->isInit()) +
172  " RHS: " + std::to_string(this->mTensorRHS->isInit()) +
173  " Output: " + std::to_string(this->mTensorOutput->isInit()));
174  }
175 
176  // TODO: Explore use-cases where tensors shouldn't be the same size, and how
177  // to deal with those situations
178  if (!(this->mTensorLHS->size() == this->mTensorRHS->size() &&
179  this->mTensorRHS->size() == this->mTensorOutput->size())) {
180  throw std::runtime_error(
181  "Kompute OpMult all tensor parameters must be the same size LHS: " +
182  std::to_string(this->mTensorLHS->size()) +
183  " RHS: " + std::to_string(this->mTensorRHS->size()) +
184  " Output: " + std::to_string(this->mTensorOutput->size()));
185  }
186 
187  this->mTensorOutputStaging = std::make_shared<Tensor>(
188  this->mTensorOutput->data(), Tensor::TensorTypes::eStaging);
189 
190  this->mTensorOutputStaging->init(
191  this->mPhysicalDevice, this->mDevice, this->mCommandBuffer);
192 
193 #if RELEASE
194  std::vector<char> shaderFileData(
195  shader_data::shaders_glsl_opmult_comp_spv,
196  shader_data::shaders_glsl_opmult_comp_spv +
197  kp::shader_data::shaders_glsl_opmult_comp_spv_len);

223 void
224 OpMult<tX, tY, tZ>::record()
225 {
226  SPDLOG_DEBUG("Kompute OpMult record called");
227 
228  // Barrier to ensure the data is finished writing to buffer memory
229  this->mTensorLHS->recordBufferMemoryBarrier(
230  vk::AccessFlagBits::eHostWrite,
231  vk::AccessFlagBits::eShaderRead,
232  vk::PipelineStageFlagBits::eHost,
233  vk::PipelineStageFlagBits::eComputeShader);
234  this->mTensorRHS->recordBufferMemoryBarrier(
235  vk::AccessFlagBits::eHostWrite,
236  vk::AccessFlagBits::eShaderRead,
237  vk::PipelineStageFlagBits::eHost,
238  vk::PipelineStageFlagBits::eComputeShader);
239 
240  this->mAlgorithm->recordDispatch(this->mX, this->mY, this->mZ);
241 
242  // Barrier to ensure the shader code is executed before buffer read
243  this->mTensorLHS->recordBufferMemoryBarrier(
244  vk::AccessFlagBits::eShaderWrite,
245  vk::AccessFlagBits::eTransferRead,
246  vk::PipelineStageFlagBits::eComputeShader,
247  vk::PipelineStageFlagBits::eTransfer);
248  this->mTensorOutput->recordBufferMemoryBarrier(
249  vk::AccessFlagBits::eShaderWrite,
250  vk::AccessFlagBits::eTransferRead,
251  vk::PipelineStageFlagBits::eComputeShader,
252  vk::PipelineStageFlagBits::eTransfer);
253 
254  this->mTensorOutputStaging->recordCopyFrom(this->mTensorOutput);
255 
256  // Buffer to ensure wait until data is copied to staging buffer
257  this->mTensorLHS->recordBufferMemoryBarrier(
258  vk::AccessFlagBits::eTransferWrite,
259  vk::AccessFlagBits::eHostRead,
260  vk::PipelineStageFlagBits::eTransfer,
261  vk::PipelineStageFlagBits::eHost);
262  this->mTensorOutput->recordBufferMemoryBarrier(
263  vk::AccessFlagBits::eTransferWrite,
264  vk::AccessFlagBits::eHostRead,
265  vk::PipelineStageFlagBits::eTransfer,
266  vk::PipelineStageFlagBits::eHost);
