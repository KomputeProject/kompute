216 {
217  SPDLOG_DEBUG("Kompute OpAlgoBase record called");
218 
219  // Barrier to ensure the data is finished writing to buffer memory
220  for (std::shared_ptr<Tensor> tensor : this->mTensors) {
221  tensor->recordBufferMemoryBarrier(
222  vk::AccessFlagBits::eHostWrite,
223  vk::AccessFlagBits::eShaderRead,
224  vk::PipelineStageFlagBits::eHost,
225  vk::PipelineStageFlagBits::eComputeShader);
226  }
227 
228  this->mAlgorithm->recordDispatch(this->mX, this->mY, this->mZ);
229 
230  if (this->mCopyOutputData) {
231  // Barrier to ensure the shader code is executed before buffer read
232  for (std::shared_ptr<Tensor> tensor : this->mTensors) {
233  tensor->recordBufferMemoryBarrier(
234  vk::AccessFlagBits::eShaderWrite,
235  vk::AccessFlagBits::eTransferRead,
236  vk::PipelineStageFlagBits::eComputeShader,
237  vk::PipelineStageFlagBits::eTransfer);
238  }
239 
240  // Record copy from and create barrier for STAGING tensors
241  for (size_t i = 0; i < this->mTensors.size(); i++) {
242  this->mOutputStagingTensors[i]->recordCopyFrom(
243  this->mTensors[i], true);
244  }
