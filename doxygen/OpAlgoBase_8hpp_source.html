278 void
279 OpAlgoBase<tX, tY, tZ>::record()
280 {
281  SPDLOG_DEBUG("Kompute OpAlgoBase record called");
282 
283  // Barrier to ensure the data is finished writing to buffer memory
284  for (std::shared_ptr<Tensor> tensor : this->mTensors) {
285  tensor->recordBufferMemoryBarrier(
286  this->mCommandBuffer,
287  vk::AccessFlagBits::eHostWrite,
288  vk::AccessFlagBits::eShaderRead,
289  vk::PipelineStageFlagBits::eHost,
290  vk::PipelineStageFlagBits::eComputeShader);
291  }
292 
293  this->mAlgorithm->recordDispatch(this->mX, this->mY, this->mZ);
294 
295  if (this->mCopyOutputData) {
296  // Barrier to ensure the shader code is executed before buffer read
297  for (const std::shared_ptr<Tensor>& tensor : this->mTensors) {
298  tensor->recordBufferMemoryBarrier(
299  this->mCommandBuffer,
300  vk::AccessFlagBits::eShaderWrite,
301  vk::AccessFlagBits::eTransferRead,
302  vk::PipelineStageFlagBits::eComputeShader,
303  vk::PipelineStageFlagBits::eTransfer);
304  }
305 
306  // Record copy from and create barrier for STAGING tensors
307  for (size_t i = 0; i < this->mTensors.size(); i++) {
308  this->mOutputStagingTensors[i]->recordCopyFrom(
309  this->mCommandBuffer,
310  this->mTensors[i],
311  true);
312  }
313  }
